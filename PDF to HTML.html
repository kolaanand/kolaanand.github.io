23.SEVEN SEGMENT DISPLAY
// 7-segment display codes for digits 0-9 (Common Anode
Inverted)
unsigned char segment_code[] = {0xC0, 0xF9, 0xA4, 0xB0,
0x99,
0x92, 0x82, 0xF8, 0x80, 0x90};
// Delay function
void delay(unsigned int ms) {
unsigned int i, j;
for (i = 0; i < ms; i++)
for (j = 0; j < 1275; j++); // Approximate 1ms delay
}
void main() {
int i;
while (1) { // Infinite loop
for (i = 0; i < 10; i++) { // Display digits 0-9
P2 = segment_code[i]; // Send segment code to Port 2
delay(1000); // Delay for 1 second
}
}
}
